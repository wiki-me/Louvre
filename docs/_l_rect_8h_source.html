191  inline bool clip(const LRectTemplate &rect)
192  {
193  TA x0 = x();
194  TA x1 = x0 + w();
195  TA y0 = y();
196  TA y1 = y0 + h();
197 
198  TA rx1 = rect.x() + rect.w();
199  TA ry1 = rect.y() + rect.h();
200 
201  // X
202  if(rect.x() > x0)
203  x0 = rect.x();
204  else if(rx1 < x0)
205  x0 = rx1;
206 
207  // W
208  if(rect.x() > x1)
209  x1 = rect.x();
210  else if(rx1 < x1)
211  x1 = rx1;
212 
213  // Y
214  if(rect.y() > y0)
215  y0 = rect.y();
216  else if(ry1 < y0)
217  y0 = ry1;
218 
219  // H
220  if(rect.y() > y1)
221  y1 = rect.y();
222  else if(ry1 < y1)
223  y1 = ry1;
224 
225  setX(x0);
226  setY(y0);
227  setW(x1 - x0);
228  setH(y1 - y0);
229 
230  return (w() == 0 || h() == 0 );
231  }
